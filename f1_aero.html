# F1 Aero Mini-Project
"""
This notebook builds a simplified aerodynamic model of a Formula 1 front wing under the 2026 regulations.

## Objectives
- Show how downforce and drag vary with speed and AoA.
- Produce clear plots:
    - Downforce vs Speed at selected AoAs
    - Drag vs Downforce trade-off curves at selected speeds
- Include a mini-optimiser that finds the AoA giving minimum drag while still meeting a target downforce requirement.

## Assumptions
- Wing reference area: 0.618 m^2 (approx. 2026 front wing planform).
- Lift curve slope: 6.0 per rad (based on thin airfoil theory, adjusted because of the front wings are viscous, cambered and
    multi-element so their slope is not exactly 2pi=6.3 per rad).
- Stall angle: 16 (degrees), modelled with smooth tanh roll-off.
- Ground effect: bounded multiplier that increases lift and drag as ride height decreases.
- Yaw penalty: reduces effective lift (cos ) and adds drag.
- Reynolds correction: mild scaling relative to Re at 30 m/s, chord length 0.35 m.

## Usage
1. Open this notebook in Jupyter.
2. Run all cells - outputs are saved to `./aero_plots/` and displayed inline.
3. Adjust parameters (ride height, yaw angle, target downforce) in the code to explore design trade-offs.

## Why This Matters
This project demonstrates:
- Connecting aero concepts (AoA, ground effect, yaw) to performance plots.
- Answering a design question: "What AoA gives enough downforce with minimum drag?"

## Background information
 Clear physics: L = 0.5  V^2 S CL(,h,,Re), D = 0.5  V^2 S CD(CL,Re)
 Small but meaningful F1-relevant effects:
- Ground effect via ride-height multiplier M_h(h)
- Yaw () penalty/redistribution
- Reynolds-number scaling (soft correction)
 Two core plots + a design table:
1) Downforce vs Speed (selected AoA)
2) Drag vs Downforce trade-offs (AoA swept at fixed speeds)
3) Optimized AoA for target downforce (min-drag solution) per speed

Assumptions:
 CL() = CL0 + aL _smooth, _smooth = _stall tanh(/_stall) [ in rad]
 CD = CD0 + k CL^2 (parabolic polar)
 Ground effect multiplier M_h(h) = 1 + A / (1 + h/h0) (bounded growth)
 Yaw adjustment: CL = CL cos , CD = CD + C (1 - cos )
 Reynolds scaling (soft): CL, CD scale by (Re/Re_ref)^m_cl and m_cd (|m|1)

Units
 SI throughout. Forces in N, speeds in m/s, angles: input/output in degrees; model uses radians.
 "Downforce" is reported as magnitude |L| (positive).

Author: [Qamer Pulikal] - Mechanical Engineering Student, QMUL
"""

try:
    import numpy as np
    import matplotlib.pyplot as plt
except Exception:
    import sys, subprocess
    subprocess.check_call([sys.executable, "-m", "pip", "install", "numpy", "matplotlib"])
    import numpy as np
    import matplotlib.pyplot as plt

from pathlib import Path

# --- Parameters
# Environment
rho = 1.225 # air density [kg/m^3]
nu = 1.5e-5 # kinematic viscosity of air [m^2/s] (ISA ~ 15degC)
S = 0.618 # approx 2026 front-wing planform area [m^2]
c_ref = 0.35 # characteristic chord [m] (order-of-magnitude for multi-element section)

# Baseline aerodynamic model
CL0 = 0.15 # CL at 0deg AoA
aL = 6.0 # lift curve slope [1/rad] (thin airfoil ~ 2  6.283)
CD0 = 0.22 # zero-lift drag coefficient
k = 0.85 # quadratic polar coefficient

# Stall softening
stall_deg = 16.0
stall_rad = np.deg2rad(stall_deg)

# Ground effect (simple, bounded gain as h --> small)
# M_h(h) = 1 + A / (1 + h/h0). Choose mild A so it doesn't explode.
A_ge = 0.25 # amplitude of ground-effect lift gain
h0_ge = 0.07 # scale height [m] ~ 70 mm (tunable)
# Also add a small drag growth when close to ground:
A_ge_D = 0.06

# Yaw effect parameters ( in radians)
C_beta = 0.10 # extra drag due to yaw

# Reynolds-number soft scaling
Re_ref = (30.0 * c_ref / nu) # reference at 30 m/s
m_cl = 0.06 # CL scale exponent (small)
m_cd = 0.02 # CD scale exponent (very small)

# Grids & selections
aoa_deg_grid = np.linspace(-5.0, 20.0, 126) # for trade-off curves
V_grid = np.linspace(10.0, 90.0, 161) # downforce vs speed
aoa_show_deg = [0, 6, 12, 16] # curves in plot 1
speeds_show_ms = [30, 50, 70] # curves in plot 2

# Operating condition knobs (F1-relevant)
h_ride = 0.070 # ride height [m] under the front wing (70 mm)
beta_deg = 3.0 # small yaw angle [deg] representative of cornering
beta_rad = np.deg2rad(beta_deg)

# Optimisation target (engineering design ask)
target_downforce_N = 3500.0 # chose a realistic per-front-wing target force [N]

# Output
out_dir = Path("./aero_plots")
out_dir.mkdir(parents=True, exist_ok=True)

# --- Core model helpers
def smooth_alpha(alpha_rad: np.ndarray, alpha_stall_rad: float) -> np.ndarray:
    """Smooth AoA near stall: _smooth = _stall * tanh(/_stall)."""
    return alpha_stall_rad * np.tanh(alpha_rad / max(1e-9, alpha_stall_rad))

def reynolds_number(V: np.ndarray, c: float = c_ref, nu_: float = nu) -> np.ndarray:
    """Chord-based Reynolds number Re = V c / ."""
    return np.asarray(V) * c / nu_

def re_scale(Re: np.ndarray, Re_ref: float = Re_ref, mcl: float = m_cl, mcd: float = m_cd) -> tuple[np.ndarray, np.ndarray]:
    """Return (phi_cl, phi_cd) scaling multipliers for CL and CD."""
    r = np.maximum(Re, 1.0) / max(Re_ref, 1.0)
    return r**mcl, r**mcd

def ground_effect_multipliers(h: float) -> tuple[float, float]:
    """
    Return (M_cl, M_cd) multipliers due to ground effect at ride height h.
    Bounded, monotone: increases lift, slightly increases drag near ground.
    """
    gain = A_ge / (1.0 + h / max(1e-6, h0_ge))
    M_cl = 1.0 + gain
    M_cd = 1.0 + A_ge_D * gain
    return M_cl, M_cd

def yaw_adjust(CL: np.ndarray, CD: np.ndarray, beta: float) -> tuple[np.ndarray, np.ndarray]:
    """
    Simple yaw model:
    CL = CL * cos  (effective lift reduction)
    CD = CD + C (1 - cos ) (added crossflow/sideforce-induced drag)
    """
    cosb = np.cos(beta)
    CLb = CL * cosb
    CDb = CD + C_beta * (1.0 - cosb)
    return CLb, CDb

def lift_coefficient(alpha_deg: np.ndarray, V: np.ndarray, h: float, beta: float) -> np.ndarray:
    """CL including stall softening, Re scaling, ground effect, and yaw."""
    alpha_rad = np.deg2rad(alpha_deg)
    alpha_eff = smooth_alpha(alpha_rad, stall_rad)
    CL_base = CL0 + aL * alpha_eff

    # Apply Reynolds scaling
    Re = reynolds_number(V)
    phi_cl, _ = re_scale(Re)
    
    # Apply ground effect
    M_cl, _ = ground_effect_multipliers(h)
    
    CL = CL_base * phi_cl * M_cl
    
    # Yaw adjustment
    CLb, _ = yaw_adjust(CL, np.zeros_like(CL), beta)
    return CLb

def drag_coefficient(CL: np.ndarray, V: np.ndarray, h: float, beta: float) -> np.ndarray:
    """CD including polar, Re scaling, ground effect drag growth, and yaw drag add-on."""
    Re = reynolds_number(V)
    _, phi_cd = re_scale(Re)
    _, M_cd = ground_effect_multipliers(h)
    
    CD_base = (CD0 * phi_cd * M_cd) + k * (CL**2)
    
    # Yaw drag add-on
    _, CDb = yaw_adjust(np.zeros_like(CL), CD_base, beta)
    return CDb

def aero_forces(CL: np.ndarray, CD: np.ndarray, V: np.ndarray) -> tuple[np.ndarray, np.ndarray]:
    """Compute (downforce magnitude, drag) in Newtons."""
    q = 0.5 * rho * (np.asarray(V)**2)
    L = q * S * CL
    D = q * S * CD
    DF = np.abs(L)
    return DF, D

# --- Plot style helpers
def _prep_axes(ax, xlabel: str, ylabel: str, title: str):
    ax.set_xlabel(xlabel); ax.set_ylabel(ylabel); ax.set_title(title)
    ax.grid(True, which="both", alpha=0.25)
    for s in ax.spines.values():
        s.set_alpha(0.4)

# --- 1) Downforce vs Speed (selected AoA)
def plot_downforce_vs_speed(aoa_list_deg, V, h, beta) -> plt.Figure:
    fig, ax = plt.subplots(figsize=(8, 5.2), dpi=150)
    for a in aoa_list_deg:
        # vectorised across V
        CL = lift_coefficient(np.full_like(V, a, dtype=float), V, h, beta)
        CD = drag_coefficient(CL, V, h, beta)
        DF, _ = aero_forces(CL, CD, V)
        ax.plot(V, DF, label=f"AoA = {a:.0f}", linewidth=2)
    _prep_axes(ax, "Speed V [m/s]", "Downforce |L| [N]",
        f"Downforce vs Speed  h={h*1e3:.0f} mm, ={np.rad2deg(beta):.1f}")
    ax.legend(title="Curves", frameon=False)
    fig.tight_layout()
    fig.savefig(out_dir / "downforce_vs_speed.png", bbox_inches="tight")
    return fig

# --- 2) Drag vs Downforce trade-off (AoA swept)
def plot_drag_vs_downforce_tradeoff(speeds_ms, aoa_deg, h, beta) -> plt.Figure:
    fig, ax = plt.subplots(figsize=(8, 5.2), dpi=150)
    for Vs in speeds_ms:
        # vectorised across AoA
        Vv = np.full_like(aoa_deg, Vs, dtype=float)
        CL = lift_coefficient(aoa_deg, Vv, h, beta)
        CD = drag_coefficient(CL, Vv, h, beta)
        DF, D = aero_forces(CL, CD, Vv)
        ax.plot(D, DF, label=f"V = {Vs:.0f} m/s", linewidth=2)
    _prep_axes(ax, "Drag D [N]", "Downforce |L| [N]",
        f"Drag vs Downforce (AoA swept)  h={h*1e3:.0f} mm, ={np.rad2deg(beta):.1f}")
    ax.legend(title="Speeds", frameon=False)
    fig.tight_layout()
    fig.savefig(out_dir / "drag_vs_downforce_tradeoff.png", bbox_inches="tight")
    return fig

# --- 3) Mini-optimiser: min-drag AoA to hit target downforce
def optimise_aoa_for_target(Vs_list, target_DF, h, beta, aoa_bounds=(-5.0, 20.0), n_grid=1001):
    """
    For each speed in Vs_list, sweep AoA within bounds and choose AoA that:
    (a) Achieves DF > target_DF (closest above target), and
    (b) Minimises drag D among feasible points.
    Returns a list of with results (speed, AoA*, DF, D).
    """
    a_min, a_max = aoa_bounds
    aoa_grid = np.linspace(a_min, a_max, n_grid)
    results = []
    
    for Vs in Vs_list:
        Vv = np.full_like(aoa_grid, Vs, dtype=float)
        CL = lift_coefficient(aoa_grid, Vv, h, beta)
        CD = drag_coefficient(CL, Vv, h, beta)
        DF, D = aero_forces(CL, CD, Vv)
        
        feasible = DF >= target_DF
        if not np.any(feasible):
            # Not reachable: pick AoA giving max DF; report as unreachable
            idx = int(np.argmax(DF))
            results.append({
                "V_ms": Vs, "reachable": False,
                "AoA_deg": float(aoa_grid[idx]),
                "Downforce_N": float(DF[idx]), "Drag_N": float(D[idx])
            })
        else:
            # Among feasible, choose minimum drag
            idxs = np.where(feasible)[0]
            j = idxs[np.argmin(D[feasible])]
            results.append({
                "V_ms": Vs, "reachable": True,
                "AoA_deg": float(aoa_grid[j]),
                "Downforce_N": float(DF[j]), "Drag_N": float(D[j])
            })
    return results

def print_design_table(opt_results, target_DF):
    print("\n=== Min-Drag AoA to meet target downforce ===")
    print(f"Target downforce: {target_DF:.0f} N")
    print(f"(h = {h_ride*1e3:.0f} mm,  = {beta_deg:.1f}, S = {S:.3f} m^2, c = {c_ref:.3f} m)")
    print("-"*70)
    print(f"{'V [m/s]':>8} {'Reach?':>7} {'AoA* [deg]':>10} {'DF [N]':>10} {'D [N]':>10}")
    for r in opt_results:
        print(f"{r['V_ms']:8.1f} {str(r['reachable']):>7} {r['AoA_deg']:10.2f} "
            f"{r['Downforce_N']:10.0f} {r['Drag_N']:10.0f}")
